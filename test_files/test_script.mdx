/* Rolling 12-month average */
WITH
MEMBER [Measures].[Rolling Unit Sales] AS
'Avg (
  LastPeriods (
    12,
    [Time].CurrentMember
  ),
  [Measures].[Unit Sales]
)'
SELECT
{ [Measures].[Unit Sales], [Measures].[Rolling Unit Sales] }
on columns,
{ 
  [Time].[Q1, 1999].Children
}
on rows
FROM [WM2K.Sales]
WHERE [Product].[Toys]


/* Year-to-year comparison */
WITH
MEMBER [Measures].[Year Ago Units Change] AS
'[Measures].[Unit Sales]
- ([Measures].[Unit Sales], 
   ParallelPeriod (
    [Time].[Year],
    1,
    [Time].CurrentMember
   )
  )'
SELECT
{ [Measures].[Unit Sales], [Measures].[Year Ago Units Change] }
on columns,
{ [Time].[1999],
  [Time].[Q1, 1999],
  [Time].[Q1, 1999].Children
}
on rows
FROM [WM2K.Sales]
WHERE [Product].[Toys]
